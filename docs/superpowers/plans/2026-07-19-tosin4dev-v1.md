# Tosin4dev v1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** A local-first multi-board Kanban console where approved tickets run autonomously through Claude/Codex CLI runners to a review-ready state, with human gates only at spec approval and final review.

**Architecture:** TanStack Start serves front + back (server functions are the API). Official `mongodb` driver + Zod (no ORM) against a local containerized Mongo. Runner adapters are pure command builders (`claude -p`, `codex exec`); a shared supervisor spawns via `Bun.spawn`, isolates execute runs in git worktrees, transitions tickets, and posts Discord pings on `blocked`/`review_ready`.

**Tech Stack:** Bun, TanStack Start (React 19), react-query-kit, Tailwind 4, mongodb, Zod, Vitest, Docker (mongo).

Spec: `docs/superpowers/specs/2026-07-19-tosin4dev-design.md` (authoritative for domain model, state machine, routes).

```ts
import { z } from "zod";
import { TicketStatus } from "./schemas";

type Status = z.infer<typeof TicketStatus>;

export const HUMAN_GATES: readonly Status[] = ["spec_review", "review_ready"];

const TABLE: Record<string, Status> = {
  "inbox:submit_spec": "spec_review",
  "spec_review:approve_spec": "approved",
  "spec_review:request_spec_changes": "inbox",
  "approved:dispatch": "running",
  "running:run_succeeded": "review_ready",
  "running:run_failed": "blocked",
  "blocked:resume": "approved",
  "review_ready:approve_final": "done",
  "review_ready:request_changes": "running",
};

export type Event =
  | "submit_spec" | "approve_spec" | "request_spec_changes"
  | "dispatch" | "run_succeeded" | "run_failed"
  | "resume" | "approve_final" | "request_changes" | "archive";

export function transition(from: Status, event: Event): Status {
  if (event === "archive") return "archived";
  const to = TABLE[`${from}:${event}`];
  if (!to) throw new Error(`invalid transition: ${from} + ${event}`);
  return to;
}
```

- [ ] **Step 2: Write `src/server/tickets.ts`**

```ts
import { createServerFn } from "@tanstack/react-start";
import { z } from "zod";
import { db, ObjectId } from "./db";
import { TicketSchema, SpecSchema } from "../domain/schemas";
import { transition, type Event } from "../domain/stateMachine";

const now = () => new Date().toISOString();

async function log(ticketId: string, kind: string, message: string) {
  await (await db()).collection("tickets").updateOne(
    { _id: new ObjectId(ticketId) },
    { $push: { activity: { $each: [{ at: now(), kind, message }], $slice: -50 } } },
  );
}

export const listTickets = createServerFn({ method: "GET" })
  .validator(z.object({ boardId: z.string() }))
  .handler(async ({ data }) => {
    return (await db()).collection("tickets")
      .find({ boardId: data.boardId }).sort({ seq: -1 }).toArray();
  });

export const getTicket = createServerFn({ method: "GET" })
  .validator(z.object({ boardId: z.string(), seq: z.number() }))
  .handler(async ({ data }) => {
    return (await db()).collection("tickets").findOne(data);
  });

export const createTicket = createServerFn({ method: "POST" })
  .validator(TicketSchema.omit({ seq: true, status: true }))
  .handler(async ({ data }) => {
    const tickets = (await db()).collection("tickets");
    const last = await tickets.find({ boardId: data.boardId })
      .sort({ seq: -1 }).limit(1).next();
    const seq = (last?.seq ?? 0) + 1;
    const doc = { ...data, seq, status: "inbox", createdAt: now(), updatedAt: now() };
    const r = await tickets.insertOne(doc);
    return { id: r.insertedId.toString(), seq };
  });

export const updateSpec = createServerFn({ method: "POST" })
  .validator(z.object({ ticketId: z.string(), spec: SpecSchema }))
  .handler(async ({ data }) => {
    await (await db()).collection("tickets").updateOne(
      { _id: new ObjectId(data.ticketId) },
      { $set: { spec: data.spec, updatedAt: now() } },
    );
    await log(data.ticketId, "spec", "spec updated");
  });

export const transitionTicket = createServerFn({ method: "POST" })
  .validator(z.object({ ticketId: z.string(), event: z.string(), prUrl: z.string().url().optional() }))
  .handler(async ({ data }) => {
    const tickets = (await db()).collection("tickets");
    const t = await tickets.findOne({ _id: new ObjectId(data.ticketId) });
    if (!t) throw new Error("ticket not found");
    const to = transition(t.status, data.event as Event);
    const $set: Record<string, unknown> = { status: to, updatedAt: now() };
    if (data.event === "approve_spec") $set["spec.approvedAt"] = now();
    if (data.prUrl) $set.prUrl = data.prUrl;
    await tickets.updateOne({ _id: t._id }, { $set });
    await log(data.ticketId, "transition", `${t.status} --${data.event}--> ${to}`);
    return { status: to };
  });
```

```ts
import type { RunnerBrief } from "./types";

export function buildPrompt(b: RunnerBrief): string {
  const { ticket, board, workDir } = b;
  const acc = ticket.spec.acceptance.map((a, i) => `${i + 1}. ${a}`).join("\n");
  const links = ticket.spec.links.length ? ticket.spec.links.join("\n") : "none";
  if (b.phase === "spec_draft") {
    return [
      `You are drafting the executable spec for ticket #${ticket.seq}: ${ticket.title}.`,
      `Repo: ${board.repoPath} (base branch: ${board.defaultBaseBranch}).`,
      `Intent: ${ticket.spec.intent}`,
      `Investigate the repo READ-ONLY and produce: a concrete plan, affected files, verification commands, and risks. Do not modify any file.`,
      `Acceptance criteria:\n${acc || "none provided"}`,
      `End your output with a section titled SUMMARY containing at most 10 lines.`,
    ].join("\n\n");
  }
  return [
    `Implement ticket #${ticket.seq}: ${ticket.title}.`,
    `Work directory (isolated git worktree): ${workDir}`,
    `Intent: ${ticket.spec.intent}`,
    `Scope (only touch these): ${ticket.spec.scope || "unspecified — stay minimal"}`,
    `Non-goals (must NOT change): ${ticket.spec.nonGoals || "none"}`,
    `Acceptance criteria:\n${acc || "none provided"}`,
    `Links:\n${links}`,
    `Rules: stay inside the worktree; run the repo's own verification commands; commit your changes on the current branch; do not push; do not open PRs.`,
    `End your output with a section titled SUMMARY containing at most 10 lines including verification results.`,
  ].join("\n\n");
}
```

```ts
import { db, ObjectId } from "./db";
import { transition } from "../domain/stateMachine";
import { buildPrompt } from "../runners/brief";
import { claudeAdapter } from "../runners/claude";
import { codexAdapter } from "../runners/codex";
import { notify } from "./notify";
import { mkdir, writeFile, appendFile } from "node:fs/promises";
import { execFile } from "node:child_process";
import { promisify } from "node:util";

const exec = promisify(execFile);
const now = () => new Date().toISOString();
const adapters = { claude: claudeAdapter, codex: codexAdapter };

export async function dispatchRun(ticketId: string, phase: "execute" | "spec_draft" | "review_fix") {
  const database = await db();
  const ticket = await database.collection("tickets").findOne({ _id: new ObjectId(ticketId) });
  if (!ticket) throw new Error("ticket not found");
  if (ticket.activeRunId) throw new Error("ticket already has an active run");
  const board = await database.collection("boards").findOne({ _id: new ObjectId(ticket.boardId) });
  if (!board) throw new Error("board not found");

  const runId = new ObjectId();
  const root = `${board.repoPath}/.tosin4dev`;
  const workDir = phase === "spec_draft"
    ? board.repoPath
    : `${root}/worktrees/${runId.toString()}`;
  const runDir = `${root}/runs/${runId.toString()}`;
  await mkdir(runDir, { recursive: true });

  if (phase !== "spec_draft") {
    await exec("git", ["-C", board.repoPath, "worktree", "add", "--detach", workDir, board.defaultBaseBranch]);
  }

  const prompt = buildPrompt({ ticket: ticket as never, board: board as never, workDir, phase });
  const promptFile = `${runDir}/prompt.md`;
  const logFile = `${runDir}/output.log`;
  await writeFile(promptFile, prompt);

  const adapter = adapters[ticket.runner as "claude" | "codex"];
  const { cmd, env } = adapter.buildCommand(
    { ticket: ticket as never, board: board as never, workDir, phase }, promptFile);

  await database.collection("runs").insertOne({
    _id: runId, ticketId, boardId: ticket.boardId, runner: ticket.runner,
    phase, status: "running", workDir, promptFile, logFile,
    exitCode: null, summary: null, startedAt: now(), finishedAt: null,
  });
  await database.collection("tickets").updateOne(
    { _id: ticket._id },
    { $set: { activeRunId: runId.toString(), status: transition(ticket.status, "dispatch"), updatedAt: now() } },
  );

  const child = Bun.spawn(cmd, {
    cwd: workDir,
    env: { ...process.env, ...env },
    stdout: "pipe", stderr: "pipe",
  });

  void (async () => {
    const out = await new Response(child.stdout).text();
    const err = await new Response(child.stderr).text();
    await appendFile(logFile, out + (err ? `\n--- STDERR ---\n${err}` : ""));
    const exitCode = await child.exited;
    await finishRun(runId.toString(), ticketId, exitCode, out);
  })();

  return { runId: runId.toString() };
}

async function finishRun(runId: string, ticketId: string, exitCode: number, output: string) {
  const database = await db();
  const ticket = await database.collection("tickets").findOne({ _id: new ObjectId(ticketId) });
  if (!ticket) return;
  const summary = output.split(/\nSUMMARY\n/i)[1]?.trim().split("\n").slice(0, 10).join("\n") ?? null;
  const ok = exitCode === 0;
  const to = transition(ticket.status, ok ? "run_succeeded" : "run_failed");
  await database.collection("runs").updateOne(
    { _id: new ObjectId(runId) },
    { $set: { status: ok ? "succeeded" : "failed", exitCode, summary, finishedAt: now() } },
  );
  await database.collection("tickets").updateOne(
    { _id: ticket._id },
    { $set: { status: to, activeRunId: null, updatedAt: now() },
      $push: { activity: { $each: [{ at: now(), kind: "run",
        message: `run ${ok ? "succeeded" : `failed (exit ${exitCode})`}` }], $slice: -50 } } },
  );
  const board = await database.collection("boards").findOne({ _id: new ObjectId(ticket.boardId) });
  const label = `${board?.slug ?? "?"} #${ticket.seq} ${ticket.title}`;
  if (to === "review_ready") await notify(`✅ review-ready: ${label}\n${summary ?? ""}`);
  if (to === "blocked") await notify(`⛔ blocked: ${label} — run failed (exit ${exitCode}). Log: ${(await database.collection("runs").findOne({ _id: new ObjectId(runId) }))?.logFile}`);
}

export async function recoverOrphans() {
  const database = await db();
  const stale = await database.collection("runs").find({ status: "running" }).toArray();
  for (const run of stale) {
    const alive = run.pid ? await exec("kill", ["-0", String(run.pid)]).then(() => true).catch(() => false) : false;
    if (!alive) {
      await database.collection("runs").updateOne({ _id: run._id },
        { $set: { status: "failed", exitCode: null, finishedAt: now() } });
      const t = await database.collection("tickets").findOne({ _id: new ObjectId(run.ticketId) });
      if (t?.activeRunId === run._id.toString()) {
        await database.collection("tickets").updateOne({ _id: t._id },
          { $set: { status: "blocked", activeRunId: null, updatedAt: now() } });
      }
    }
  }
}
```

```tsx
import { useState } from "react";
import { useRuns, useDispatch, useLogTail } from "../queries/runs";

export function RunsSection({ ticket }: { ticket: any }) {
  const { data: runs = [] } = useRuns({ variables: { ticketId: ticket._id.toString() } });
  const dispatchRun = useDispatch();
  const [openRun, setOpenRun] = useState<string | null>(null);
  const { data: log } = useLogTail(
    openRun ? { variables: { runId: openRun } } : (undefined as never),
    { refetchInterval: 2000 },
  );

  return (
    <section className="space-y-3">
      <div className="flex items-center justify-between">
        <h2 className="font-semibold">Runs</h2>
        <div className="flex gap-2">
          {ticket.status === "approved" && (
            <button className="rounded bg-zinc-900 px-3 py-1.5 text-sm text-white"
              onClick={() => dispatchRun.mutate({ ticketId: ticket._id.toString(), phase: "execute" })}>
              Run now
            </button>
          )}
          {ticket.status === "inbox" && (
            <button className="rounded bg-zinc-700 px-3 py-1.5 text-sm text-white"
              onClick={() => dispatchRun.mutate({ ticketId: ticket._id.toString(), phase: "spec_draft" })}>
              Draft spec with {ticket.runner}
            </button>
          )}
        </div>
      </div>
      <ul className="space-y-1 text-sm">
        {runs.map((r: any) => (
          <li key={r._id} className="flex items-center gap-3 rounded border p-2">
            <span className="font-mono text-xs">{r.phase}</span>
            <span>{r.status}</span>
            <span className="text-xs text-zinc-500">{r.startedAt?.slice(0, 19).replace("T", " ")}</span>
            <button className="ml-auto text-blue-700 underline text-xs"
              onClick={() => setOpenRun(openRun === r._id.toString() ? null : r._id.toString())}>
              {openRun === r._id.toString() ? "hide log" : "log"}
            </button>
          </li>
        ))}
      </ul>
      {openRun && (
        <pre className="max-h-96 overflow-auto rounded bg-zinc-950 p-3 text-xs text-zinc-100">
          {log?.text || "…"}
        </pre>
      )}
    </section>
  );
}
```
